const USERS_KEY = "users"

/**
 * Obtiene la lista de usuarios almacenada en el local storage
 * Si no hay usuarios almacenados, se crea una lista inicial con un usuario administrador
 * @returns {Array} Una lista de objetos con los datos de los usuarios
 */
export function getUsers() {
  let usersString = localStorage.getItem(USERS_KEY)
  if (!usersString) {
    const initialValue = [{name: "Frankz Alarcon", phone: "0987654321", email: "frankz@correo.com", password: "12345678", role: "admin"}]    
    localStorage.setItem(USERS_KEY, JSON.stringify(initialValue))
    return initialValue
  }
  const users = JSON.parse(usersString)  
  return users
}

/**
 * Agrega un nuevo usuario a la lista almacenada en el local storage
 * @param {Object} user - Un objeto con los datos del usuario a agregar
 * @returns {Array} La lista actualizada de usuarios, incluyendo el nuevo usuario agregado
 */
export function addUser(user) {
  const users = getUsers()
  const newUsers = [...users, user]
  localStorage.setItem(USERS_KEY, JSON.stringify(newUsers))
  return newUsers
}
