/**
 * @function formatPrice
 * @description Formatea un número en un string que representa una cantidad monetaria.
 * @param {number} price - El precio a formatear.
 * @returns {string} - El precio formateado en una cadena de texto.
 */
export const formatPrice = (price) => {
  const x = new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency: "USD"
  }).format(price)
  return x
}

/**
 * @function formatDate
 * @description Convierte una cadena de fecha en un objeto Date y devuelve la fecha formateada.
 * @param {string} dateString - La cadena de texto que representa una fecha.
 * @returns {string} - La fecha formateada en una cadena de texto en el formato local "es-ES".
 */
export const formatDate = (dateString) => {
  const date = new Date(dateString)
  return date.toLocaleDateString('es-ES', {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
    weekday: 'long'
  })
}

/**
 * @function getCookie
 * @description Obtiene un valor de cookie por su clave.
 * @param {string} key - La clave del valor de la cookie a obtener.
 * @returns {object} - El valor de la cookie en forma de objeto JSON.
 */
export const getCookie = (key) => {
  const cookie = document.cookie.split(";")
    .find(c => c.trim().startsWith(`${key}=`))
    .split("=")[1]
  return JSON.parse(cookie)
}

/**
 * @function setCookie
 * @description Establece un valor de cookie para una clave dada.
 * @param {string} key - La clave a la que se debe asociar el valor de la cookie.
 * @param {object} value - El valor que se debe almacenar en la cookie.
 * @returns {void}
 */
export const setCookie = (key, value) => {
  document.cookie = `${key}=${JSON.stringify(value)};path=/;`
}

/**
 * @function clearCookie
 * @description Elimina una cookie existente para una clave dada.
 * @param {string} key - La clave de la cookie que se debe eliminar.
 * @returns {void}
 */

export function clearCookie(key) {
  document.cookie = `${key}=;path=/;`
}

/**
 * @function getLocalStorage
 * @description Obtiene el valor almacenado en el almacenamiento local para una clave dada.
 * @param {string} key - La clave para la cual se debe obtener el valor almacenado.
 * @returns {object} - El valor almacenado en el almacenamiento local en forma de objeto JSON.
 */
export const getLocalStorage = (key) => {
  const value = localStorage.getItem(key)
  return JSON.parse(value)
}


/**
 * @function setLocalStorage
 * @description Establece un valor en el almacenamiento local para una clave dada.
 * Si la clave ya tiene un valor asociado, se sobrescribirá con el nuevo valor proporcionado.
 * @param {string} key - La clave a la que se debe asociar el valor.
 * @param {object} value - El valor que se debe almacenar en el almacenamiento local.
 * @returns {void}
 */
export const setLocalStorage = (key, value) => {
  const data = localStorage.getItem(key)
  if (!data) {
    localStorage.setItem(key, JSON.stringify(value))
  }
  localStorage.setItem(key, JSON.stringify(value))
}
